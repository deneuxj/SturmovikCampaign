	trigger Hide
	trigger Show
}

type AntiTankCanon {
	entity Canon
}

type WhileEnemyClose from VirtualConvoy
LeadAntiTankCanon lead_of_enemy_close(WhileEnemyClose)

// LeadAntiTankCanon <-> WhileEnemyClose
object += for (lead, wec) in lead_of_enemy_close -> (wec.Proximity, lead.Canon)
